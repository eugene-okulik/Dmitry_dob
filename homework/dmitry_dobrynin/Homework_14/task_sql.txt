-- "Поставьте своему студенту оценки (marks) для всех созданных вами занятий"
INSERT INTO marks(value, lesson_id, student_id)
VALUES (4, 76240, 22936),
	   (5, 76241, 22936),
	   (3, 76242, 22936),
	   (5, 76243, 22936);

SELECT *
FROM subjects s
JOIN lessons l on s.id = l.subject_id 
JOIN marks m on l.id = m.lesson_id
JOIN students s2 on m.student_id = s2.id
WHERE s.id IN (22941, 22942);


--  "Все оценки студента"
SELECT m.value as 'Все оценки студента',
       s.name as 'Имя',
       s.second_name as 'Фамилия'
FROM marks m 
JOIN students s on s.id = m.student_id
WHERE s.id = (SELECT s2.id
              FROM students s2 
              WHERE s2.name = 'Dmitry' 
                    and s2.second_name = 'Dobrynin');


-- Дополнительно посчитал средний балл
SELECT AVG(m.value) as 'Средний балл',
       s.name as 'Имя',
       s.second_name as 'Фамилия'
FROM marks m 
JOIN students s on s.id = m.student_id
WHERE s.id = (SELECT s2.id
              FROM students s2 
              WHERE s2.name = 'Dmitry' 
                    and s2.second_name = 'Dobrynin')
GROUP BY s.id;


--  Все книги, которые находятся у студента
SELECT b.title 
FROM books b
JOIN students s on s.id = b.taken_by_student_id
WHERE s.id = (SELECT s2.id
              FROM students s2 
              WHERE s2.name = 'Dmitry' 
                    and s2.second_name = 'Dobrynin');


/* 
Для вашего студента выведите всё, что о нем есть в базе: группа, книги, 
оценки с названиями занятий и предметов (всё одним запросом с использованием Join)
*/

SELECT *
FROM students s
JOIN `groups` g on g.id = s.group_id 
JOIN books b on b.taken_by_student_id = s.id
JOIN marks m on m.student_id = s.id
JOIN lessons l on l.id = m.lesson_id
JOIN subjects s2 on s2.id = l.subject_id
WHERE s.id = 22936;
